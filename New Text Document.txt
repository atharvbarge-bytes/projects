<script>
let users = [];
let currentUser = null;

function hideAll() {
    document.querySelectorAll(".card").forEach(c => c.classList.add("hidden"));
}

function goBack() {
    hideAll();
    document.getElementById("menu").classList.remove("hidden");
}

function showRegister() {
    hideAll();
    document.getElementById("register").classList.remove("hidden");
}

function showLogin() {
    hideAll();
    document.getElementById("login").classList.remove("hidden");
}

function showBankDetails() {
    hideAll();
    document.getElementById("bank").classList.remove("hidden");
    document.getElementById("totalUsers").innerText = users.length;

    let sum = users.reduce((a, u) => a + u.balance, 0);
    document.getElementById("totalBalance").innerText = sum;
}

function registerUser() {
    let name = rname.value;
    let user = ruser.value;
    let pass = rpass.value;

    users.push({ name, user, pass, balance: 0 });
    regMsg.innerHTML = "<span class='success'>User Registered Successfully</span>";

    rname.value = ruser.value = rpass.value = "";
}

function loginUser() {
    let u = luser.value;
    let p = lpass.value;

    let found = users.find(x => x.user === u && x.pass === p);

    if (found) {
        currentUser = found;
        showDashboard();
    } else {
        loginMsg.innerHTML = "<span class='error'>Invalid Credentials</span>";
    }
}

function showDashboard() {
    hideAll();
    dashboard.classList.remove("hidden");
    userName.innerText = currentUser.name;
    balance.innerText = currentUser.balance;
}

function deposit() {
    currentUser.balance += 1000;
    balance.innerText = currentUser.balance;
}

function withdraw() {
    if (currentUser.balance >= 500) {
        currentUser.balance -= 500;
        balance.innerText = currentUser.balance;
    } else {
        alert("Insufficient Balance");
    }
}

function logout() {
    currentUser = null;
    goBack();
}
</script>
